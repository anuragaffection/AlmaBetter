Database Management System {
    -- DBMS is a software that interact with databases
    -- SQL, Structured Query Language 
    -- Sharding, horizontally partition of a database 

    Types of DBMS
    -- Relational DBMS (RDBMS),  SQL
    -- Non Relational DBMS, NoSQL
    -- Object Oreinted DBMS (OODBMS) 
    -- hierarchical DBMS 
    -- Network DBMS
    -- Parallel DBMS
    -- distributed DBMS

    3 tier architecture
    -- Presentation tier [client tier]
    -- Application tier  [logic or middle tier ]
    -- Database tier     [ data management]

}







Data Language {
    -- it refers to commands that interact with database
    -- DDL defines schema, 
    -- DML manipulates data, 
    -- DCL controls permissions, 
    -- TCL manages transactions.
    
    DDL Data Definition Language
    -- CREATE
    -- ALTER
    -- DROP
    -- TRUCATE
    -- COMMIT
    -- RENAME

    DML Data Manipulation Language
    -- INSERT
    -- UPDATE
    -- DELETE
    -- MERGE
    -- CALL
    -- EXPLAIN PLAN
    -- SELECT

    DQL Data Query Language
    -- SELECT

    DCL Data Control Language
    -- GRANT 
    -- REVOKE 

    TCL Transaction Control Language
    -- COMMIT 
    -- ROLLBACK
    -- SAVEPOINT
    -- AUTOCOMMIT 

}
